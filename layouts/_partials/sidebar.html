{{ template "sidebar-menu" (dict "ctx" site.Home "level" 0 "activePage" .) }}
{{- define "sidebar-menu" -}} {{- $ctx := .ctx -}} {{- $level := .level -}} {{- $activePage := .activePage }} {{- if gt $level 1 -}} {{- fmt.Warnf "Attempted to generate a sidebar with more levels than are currently supported" -}} {{- return -}} {{- end -}} {{- range $page := (union $ctx.RegularPages $ctx.Sections).ByWeight -}} {{- if and ($page.IsSection) (eq $page.Pages.Len 0) -}} {{- fmt.Warnf "Encountered an empty section when generating the sidebar: %v" $page -}} {{- continue -}} {{- end -}} {{- $linkTitle := partial "title.html" $page -}} {{- if $page.IsSection -}} {{- $linkTitle -}} {{- template "sidebar-menu" (dict "ctx" $page "level" (add $level 1) "activePage" $activePage) -}} {{- else -}} {{- $linkClass := "" -}} {{- if eq $activePage.RelPermalink $page.RelPermalink -}}{{- $linkClass = "active" -}}{{- end -}} {{- $linkTitle -}} {{- end }} {{- end }} {{- end -}}
